def matxMultiply(A, B):
    if A == None or B == None:
        raise Exception("A or B can not be none, please Check!")
    a_row, a_col = shape(A)
    b_row, b_col = shape(B)
    if a_col != b_row:
        raise ValueError
    matrix =  [[sum(a * b for a, b in zip(a, b)) for b in zip(*B)] for a in A]
    return matrix

""" Gaussian Jordan  Ax = b.

        epsilon: 0 1.0e-16
        
     x  Ax = b 
    None Ab 
    None A 
"""

def gj_Solve(A, b, decPts=4, epsilon=1.0e-16):
    if len(A) != len(b):
        return None
    
    #
    Ab = augmentMatrix(A, b)
    row, col = shape(Ab)
    for i in range(0, len(A)):
        c = list(zip(*Ab))[i][i:]
        max_value = 0
        max_index = 0
        for idx in range(0, len(c)):                                      #
            if (abs(c[idx]) > max_value):
                max_value = abs(c[idx])
                max_index = idx + i
        if(abs(max_value) < epsilon):                                     #
            return None
        
        swapRows(Ab, i, max_index)                                        #C
        scaleRow(Ab, i, 1.0 / Ab[i][i])                                   #C1
        for j in range(0, row):                                           #C0
            if j != i and Ab[j][i] != 0:
                addScaledRow(Ab, j, i, -Ab[j][i])                         #r + (-1*r) = 0
    
    #matxRound(Ab, decPts)
    #temp = list(zip(*Ab))
    #res = temp[len(temp) - 1]
    
    res = [c[-1:] for c in Ab]
    matxRound(res, decPts)
    return res

def calculateMSE2D(X,Y,m,b):
    if len(X) != len(Y) or len(X) == 0 :
        return
    count = 0
    for i in range(0, len(X)):
        temp = pow((Y[i] - m*X[i] - b), 2)
        count = count + temp
    res = count / len(X)
    return res
